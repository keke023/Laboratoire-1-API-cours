List of possible query string

Addition

? op = + & x = number & y = number
return { op: " ", x: "1", y: "-5", result: -4}

Substraction

? op = - & x = number & y = number
return { op: "-", x: "1", y: "1", result: 0}

Multiplication

? op = * & x = number & y = number
return { op: "*", x: "1", y: "1", result: 1}

Division

? op = / & x = number & y = number
return { op: "/", x: "1", y: "1", result: 1}

Modulo

? op = % & x = number & y = number
return { op: "%", x: "100", y: "3", result: 1}

Factorial

? op = ! & n = integer
return { op: "!", n: "5", result: 120}

isPrime

? op = p & n = integer
return { op: "!", n: "5", result: true}

findPrime

? op = np & n = integer
return { op: "!", n: "1", result: 2}
